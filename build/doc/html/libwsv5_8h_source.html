242typedef struct {
243 /* === Connection Parameters === */
244 const char *host; /* IP or hostname of OBS (e.g., "192.168.1.100" or "obs.example.com") */
245 uint16_t port; /* OBS WebSocket server port, usually 4455, sometimes 4454 for WSS */
246 const char *password; /* OBS WebSocket password from settings. Set to NULL for no auth. */
247 bool use_ssl; /* Use WSS (WebSocket Secure) instead of WS. Requires OBS configured for SSL. */
248
249 /* === Timeout Settings (all in milliseconds) ===
250 Timeouts are important to prevent hanging. Too short and you get false failures.
251 Too long and your app freezes. Adjust based on network quality. */
252 uint32_t connect_timeout_ms; /* How long to wait for initial TCP connection (default: 5000) */
253 uint32_t recv_timeout_ms; /* How long to wait for data from OBS (default: 5000) */
254 uint32_t send_timeout_ms; /* How long to wait to send data to OBS (default: 5000) */
255
256 /* === Keep-Alive / Health Monitoring ===
257 The library sends ping messages periodically to detect dead connections.
258 If OBS stops responding to pings, the library will try to reconnect. */
259 uint32_t ping_interval_ms; /* Send ping this often (default: 10000, 0 to disable pings) */
260 uint32_t ping_timeout_ms; /* Wait this long for pong response (default: 5000) */
261
262 /* === Automatic Reconnection ===
263 If the connection dies, should we try to reconnect? Very useful for production
264 because networks hiccup, OBS crashes, etc. The library uses exponential backoff
265 to avoid hammering the server - delays double each attempt up to the max. */
266 bool auto_reconnect; /* Enable automatic reconnection (default: true) */
267 uint32_t reconnect_delay_ms; /* Wait this long before first reconnect (default: 1000) */
268 uint32_t max_reconnect_delay_ms; /* Don't wait longer than this between attempts (default: 30000) */
269 uint32_t max_reconnect_attempts; /* Give up after this many attempts (0 = retry forever) */
270
271 /* === Callbacks ===
272 These optional callbacks let you be notified of important events.
273 You can leave any of them NULL if you don't care about that event type. */
274 obsws_log_callback_t log_callback; /* Called when library logs something */
275 obsws_event_callback_t event_callback; /* Called when OBS sends an event */
276 obsws_state_callback_t state_callback; /* Called when connection state changes */
277 void *user_data; /* Passed to all callbacks - use for context (like "this" pointer) */
278} obsws_config_t;

296 time_t connected_since; /* Unix timestamp of when this connection was established */
297} obsws_stats_t;
