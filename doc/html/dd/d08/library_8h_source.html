17/* ============================================================================
18 * OBS WebSocket v5 Protocol Library
19 * ============================================================================
20 * A robust and efficient implementation of the OBS WebSocket v5 protocol
21 * for managing multiple OBS connections in multi-stream setups.
22 *
23 * Features:
24 * - Connection management with automatic reconnection
25 * - OBS WebSocket v5 authentication
26 * - Scene switching and control
27 * - Connection health monitoring
28 * - Thread-safe operations
29 * - Multiple concurrent connections support

33/**
34 * Error codes returned by library functions.
35 *
36 * These error codes provide detailed information about what went wrong during
37 * library operations. Unlike generic error codes, they help distinguish between
38 * different failure modes so you can implement proper error handling and recovery
39 * strategies. For example, OBSWS_ERROR_TIMEOUT means you should probably retry
40 * the operation, while OBSWS_ERROR_AUTH_FAILED means retrying won't help - the
41 * password is just wrong.
42 *
43 * The library uses negative error codes following POSIX conventions. Zero is always
44 * success. This makes it easy to check errors with simple conditions like if (error < 0).
45 *
46 * Note that some errors are recoverable (network timeouts, temporary connection
47 * failures) while others are not (invalid parameters, authentication failure).
48 * The auto-reconnect feature only applies to network-level errors, not application
49 * errors like wrong scene names.
50 */
51typedef enum {
52 OBSWS_OK = 0,
53
54 /* Parameter validation errors (application layer) - not recoverable by retrying */
55 OBSWS_ERROR_INVALID_PARAM = -1,
56
57 /* Network-level errors (can be recovered with reconnection) */
58 OBSWS_ERROR_CONNECTION_FAILED = -2,
59 OBSWS_ERROR_SEND_FAILED = -5,
60 OBSWS_ERROR_RECV_FAILED = -6,
61 OBSWS_ERROR_SSL_FAILED = -11,
62
63 /* Authentication errors (recoverable only by fixing the password) */
64 OBSWS_ERROR_AUTH_FAILED = -3,
65
66 /* Protocol/messaging errors (typically indicate bad request data or OBS issues) */
67 OBSWS_ERROR_PARSE_FAILED = -7,
68 OBSWS_ERROR_NOT_CONNECTED = -8,
69 OBSWS_ERROR_ALREADY_CONNECTED = -9,
70
71 /* Timeout errors (recoverable by retrying with patience) */
72 OBSWS_ERROR_TIMEOUT = -4,
73
74 /* System resource errors (usually indicates system-wide issues) */
75 OBSWS_ERROR_OUT_OF_MEMORY = -10,
76
77 /* Catch-all for things we didn't expect */
78 OBSWS_ERROR_UNKNOWN = -99
79} obsws_error_t;

81/**
82 * Connection state - represents the current phase of the connection lifecycle.
83 *
84 * The connection goes through several states as it initializes. Understanding these
85 * states is important because different operations are only valid in certain states.
86 * For example, you can't send scene-switching commands when the state is CONNECTING
87 * - you have to wait until CONNECTED.
88 *
89 * The state machine looks like this:
90 * DISCONNECTED -> CONNECTING -> AUTHENTICATING -> CONNECTED
91 * Any state can transition to ERROR if something goes wrong
92 * CONNECTED or ERROR can go back to DISCONNECTED when closing
93 *
94 * When you get a state callback, it tells you the old and new states so you can
95 * react appropriately. For example, you might want to disable UI buttons when
96 * moving from CONNECTED to DISCONNECTED.
97 */
98typedef enum {
99 OBSWS_STATE_DISCONNECTED = 0, /* Not connected to OBS, no operations possible */
100 OBSWS_STATE_CONNECTING = 1, /* WebSocket handshake in progress, wait for AUTHENTICATING */
101 OBSWS_STATE_AUTHENTICATING = 2, /* Connected but still doing auth, wait for CONNECTED */
102 OBSWS_STATE_CONNECTED = 3, /* Ready - authentication complete, send commands now */
103 OBSWS_STATE_ERROR = 4 /* Unrecoverable error occurred, reconnection might help */
104} obsws_state_t;

163/**
164 * Event callback function type - called when OBS sends an event.
165 *
166 * OBS sends events to tell you about things happening - scene changed, recording
167 * started, input muted, etc. Events come as JSON in event_data. You have to parse
168 * it yourself using something like cJSON. We don't parse it for you because different
169 * applications care about different events, so we save CPU by leaving parsing to you.
170 *
171 * @param conn The connection this event came from (useful if you have multiple OBS instances)
172 * @param event_type String name of the event like "SceneChanged", "RecordingStateChanged"
173 * @param event_data JSON string with event details - parse this yourself with cJSON or json-c
174 * @param user_data Pointer you provided in the config
175 *
176 * @note The event_data buffer is temporary and freed after this callback returns.
177 * If you need to keep the data, copy the string or parse it immediately.
178 * @note This callback is called from an internal thread, so synchronize access to
179 * shared data structures.
180 * @note Don't block or do expensive work in this callback - events could pile up.
181 *
182 * @example Parsing an event might look like:
183 * cJSON *event_obj = cJSON_Parse(event_data);
184 * if (event_obj && cJSON_HasObjectItem(event_obj, "eventData")) {
185 * cJSON *data = cJSON_GetObjectItem(event_obj, "eventData");
186 * // Now examine event_obj to see what changed
187 * }
188 * if (event_obj) cJSON_Delete(event_obj);
189 */
190typedef void (*obsws_event_callback_t)(obsws_connection_t *conn, const char *event_type, const char *event_data, void *user_data);

192/**
193 * State callback function type - called when connection state changes.
194 *
195 * This is how you know when the connection comes up or goes down. Use this to
196 * update your UI - disable buttons when disconnected, enable them when connected,
197 * show spinners during connecting, etc.
198 *
199 * The callback receives both the old and new states so you can see the transition.
200 * For example, if old_state is DISCONNECTED and new_state is CONNECTING, you might
201 * show a "connecting..." message. If old_state is CONNECTED and new_state is
202 * DISCONNECTED, you might show "disconnected" and disable sending commands.
203 *
204 * @param conn The connection whose state changed
205 * @param old_state Previous state (what it was before)
206 * @param new_state Current state (what it is now)
207 * @param user_data Pointer you provided in the config
208 *
209 * @note This callback is called from an internal thread, so protect shared data.
210 * @note Don't do slow operations here - state changes should be handled quickly.
211 *
212 * State transitions:
213 * - DISCONNECTED -> CONNECTING (connection attempt starting)
214 * - CONNECTING -> AUTHENTICATING (WebSocket connected, checking auth)
215 * - AUTHENTICATING -> CONNECTED (ready to use)
216 * - CONNECTING -> ERROR (connection failed)
217 * - AUTHENTICATING -> ERROR (auth failed)
218 * - CONNECTED -> DISCONNECTED (user disconnected or connection lost)
219 * - CONNECTED -> ERROR (unexpected connection drop)
220 * - ERROR -> DISCONNECTED (after cleanup)
221 * - Any state -> DISCONNECTED (when you call obsws_disconnect)
222 */
223typedef void (*obsws_state_callback_t)(obsws_connection_t *conn, obsws_state_t old_state, obsws_state_t new_state, void *user_data);

225/**
226 * Connection configuration structure.
227 *
228 * This structure holds all the settings for connecting to OBS. You should fill this
229 * out with your specific needs, then pass it to obsws_connect(). A good starting point
230 * is to call obsws_config_init() which fills it with reasonable defaults, then only
231 * change the fields you care about (usually just host, port, and password).
232 *
233 * Design note: We use a config struct instead of many function parameters because
234 * it's more flexible - adding new configuration options doesn't break existing code.
235 * It also makes it clear what options are available.
236 */
237typedef struct {
238 /* === Connection Parameters === */
239 const char *host; /* IP or hostname of OBS (e.g., "192.168.1.100" or "obs.example.com") */
240 uint16_t port; /* OBS WebSocket server port, usually 4455, sometimes 4454 for WSS */
241 const char *password; /* OBS WebSocket password from settings. Set to NULL for no auth. */
242 bool use_ssl; /* Use WSS (WebSocket Secure) instead of WS. Requires OBS configured for SSL. */
243
244 /* === Timeout Settings (all in milliseconds) ===
245 Timeouts are important to prevent hanging. Too short and you get false failures.
246 Too long and your app freezes. Adjust based on network quality. */
247 uint32_t connect_timeout_ms; /* How long to wait for initial TCP connection (default: 5000) */
248 uint32_t recv_timeout_ms; /* How long to wait for data from OBS (default: 5000) */
249 uint32_t send_timeout_ms; /* How long to wait to send data to OBS (default: 5000) */
250
251 /* === Keep-Alive / Health Monitoring ===
252 The library sends ping messages periodically to detect dead connections.
253 If OBS stops responding to pings, the library will try to reconnect. */
254 uint32_t ping_interval_ms; /* Send ping this often (default: 10000, 0 to disable pings) */
255 uint32_t ping_timeout_ms; /* Wait this long for pong response (default: 5000) */
256
257 /* === Automatic Reconnection ===
258 If the connection dies, should we try to reconnect? Very useful for production
259 because networks hiccup, OBS crashes, etc. The library uses exponential backoff
260 to avoid hammering the server - delays double each attempt up to the max. */
261 bool auto_reconnect; /* Enable automatic reconnection (default: true) */
262 uint32_t reconnect_delay_ms; /* Wait this long before first reconnect (default: 1000) */
263 uint32_t max_reconnect_delay_ms; /* Don't wait longer than this between attempts (default: 30000) */
264 uint32_t max_reconnect_attempts; /* Give up after this many attempts (0 = retry forever) */
265
266 /* === Callbacks ===
267 These optional callbacks let you be notified of important events.
268 You can leave any of them NULL if you don't care about that event type. */
269 obsws_log_callback_t log_callback; /* Called when library logs something */
270 obsws_event_callback_t event_callback; /* Called when OBS sends an event */
271 obsws_state_callback_t state_callback; /* Called when connection state changes */
272 void *user_data; /* Passed to all callbacks - use for context (like "this" pointer) */
273} obsws_config_t;

294/**
295 * Response structure for requests to OBS.
296 *
297 * When you send a request like obsws_set_current_scene(), you can get back a
298 * response with the result. The response tells you if it succeeded, and if not,
299 * why it failed. It might also contain response data from OBS - for example,
300 * obsws_get_current_scene() puts the scene name in response_data as JSON.
301 *
302 * If you don't care about the response, you can pass NULL and not get one back.
303 * Otherwise you must free it with obsws_response_free() when done.
304 *
305 * Design note: Responses are returned as strings instead of parsed JSON to save
306 * CPU - different callers care about different response fields, so we let them
307 * parse what they need. This also avoids dependency bloat.
308 */
309typedef struct {
310 bool success; /* true if OBS said the operation worked */
311 int status_code; /* OBS status code: 100-199 = success, 600+ = error */
312 char *error_message; /* If success is false, this has the reason (e.g., "Scene does not exist") */
313 char *response_data; /* Raw JSON response from OBS - parse yourself with cJSON */
314} obsws_response_t;

435/**
436 * Create a new OBS WebSocket connection and start connecting.
437 *
438 * This function creates a connection object and begins the connection process
439 * in the background. The connection goes through states: DISCONNECTED ->
440 * CONNECTING -> AUTHENTICATING -> CONNECTED. You'll be notified of state
441 * changes via the state callback (if you provided one in config).
442 *
443 * This is non-blocking - it returns immediately. The actual connection happens
444 * in a background thread. Check obsws_get_state() to see when it reaches CONNECTED.
445 *
446 * Design note: We do connection in the background because it can take a while
447 * (DNS lookups, TCP handshake, authentication). Blocking would freeze your app.
448 *
449 * @param config Connection configuration (required, cannot be NULL)
450 * @return Connection handle on success (never NULL if called with valid config),
451 * NULL on failure (usually out of memory)
452 *
453 * @note The config structure is copied internally, so you can free or reuse it
454 * after calling obsws_connect().
455 * @note The returned handle is opaque - use it with other obsws_* functions.
456 * @note Connection attempt happens asynchronously - wait for state callback or
457 * call obsws_get_state() to check when it's ready.
458 * @note If auto_reconnect is enabled in config, the library will automatically
459 * try to reconnect if the connection drops.
460 *
461 * @example Usage:
462 * obsws_config_t config;
463 * obsws_config_init(&config);
464 * config.host = "192.168.1.100";
465 * config.port = 4455;
466 * config.password = "mypassword";
467 * config.state_callback = my_state_callback;
468 * config.user_data = my_app_context;
469 *
470 * obsws_connection_t *conn = obsws_connect(&config);
471 * if (!conn) {
472 * fprintf(stderr, "Failed to create connection\n");
473 * return 1;
474 * }
475 * // Wait for state callback to indicate CONNECTED...
476 */
477obsws_connection_t* obsws_connect(const obsws_config_t *config);

482 * This closes the connection to OBS and releases all associated resources.
483 * After calling this, the connection handle is invalid - don't use it anymore.

600 * from the old scene disappear. This is the main way to change what's being streamed
601 * or recorded.
602 *
603 * Switching scenes is fast but not instant - OBS processes the request and sends
604 * a SceneChanged event when complete. If you have many sources with animations,
605 * the transition might take a second or so.
606 *
607 * @param conn Connection handle (must be in CONNECTED state)
608 * @param scene_name Name of the scene to activate (must exist in OBS, case-sensitive)
609 * @param response Optional pointer to receive response. If you pass a pointer to
610 * response pointer, you get back a response object that you must free with
611 * obsws_response_free(). Pass NULL if you don't care about the response.
612 * @return OBSWS_OK if the request was sent successfully (doesn't mean OBS processed it yet)
613 * @return OBSWS_ERROR_NOT_CONNECTED if not connected
614 * @return OBSWS_ERROR_INVALID_PARAM if scene_name is NULL
615 *
616 * @note The scene must exist in OBS. If you try to switch to a non-existent scene,
617 * OBS will return an error in the response.
618 * @note This is async - the function returns before OBS actually switches scenes.
619 * Watch for a SceneChanged event to know when it completed.
620 * @note If you only care about success/failure, you can pass NULL for response.
621 * @note The response contains OBS status codes - 100-199 is success, 600+ is error.
622 *
623 * @example Switching to a scene:
624 * obsws_response_t *response = NULL;
625 * obsws_error_t err = obsws_set_current_scene(conn, "Gaming Scene", &response);
626 * if (err == OBSWS_OK && response) {
627 * if (response->success) {
628 * printf("Scene switched successfully\n");
629 * } else {
630 * printf("OBS error: %s\n", response->error_message);
631 * }
632 * obsws_response_free(response);
633 * }
634 */
635obsws_error_t obsws_set_current_scene(obsws_connection_t *conn, const char *scene_name, obsws_response_t **response);

955/**
956 * Send a custom request to OBS using the WebSocket protocol.
957 *
958 * This is the escape hatch for accessing any OBS WebSocket v5 API that the library
959 * doesn't have a convenience function for. You specify the request type (like
960 * "GetVersion", "SetSourceName", etc.) and optionally provide request data as a
961 * JSON string. The response comes back as JSON that you parse yourself.
962 *
963 * The library handles all the protocol overhead - request IDs, timeouts, etc.
964 * You just provide the high-level request type and parameters.
965 *
966 * Why provide this instead of wrapping everything? Because the OBS API has many
967 * functions, and we wanted to keep the library focused on the most common operations.
968 * This gives you access to everything else without bloating the library.
969 *
970 * @param conn Connection handle (must be in CONNECTED state)
971 * @param request_type OBS request type name (e.g., "GetVersion", "SetSourceName")
972 * See OBS WebSocket v5 documentation for complete list
973 * @param request_data JSON string with request parameters, or NULL if no parameters needed.
974 * For example: "{\"sceneName\": \"Main\", \"sourceName\": \"Camera\"}"
975 * @param response Pointer to receive response object (must be freed with obsws_response_free)
976 * If you pass NULL, the request is still sent but you don't get a response back.
977 * @param timeout_ms How long to wait for a response (milliseconds). 0 = use default timeout.
978 * @return OBSWS_OK if request was sent and response received within timeout
979 * @return OBSWS_ERROR_NOT_CONNECTED if not connected
980 * @return OBSWS_ERROR_TIMEOUT if no response within timeout_ms
981 * @return OBSWS_ERROR_INVALID_PARAM if request_type is NULL
982 * @return OBSWS_ERROR_PARSE_FAILED if response JSON was malformed
983 *
984 * @note You must know the OBS WebSocket v5 API to use this effectively. See
985 * https://github.com/obsproject/obs-websocket/blob/master/docs/generated/protocol.md
986 * @note The response_data field in the response contains the OBS response as a
987 * JSON string. You parse it with cJSON or similar.
988 * @note If response is NULL, this becomes fire-and-forget - useful for commands
989 * where you don't care about the result.
990 * @note Timeout of 0 uses the global receive timeout from config (usually 5000ms).
991 *
992 * @example Getting OBS version:
993 * obsws_response_t *response = NULL;
994 * obsws_error_t err = obsws_send_request(conn, "GetVersion", NULL, &response, 5000);
995 * if (err == OBSWS_OK && response) {
996 * // Parse response->response_data as JSON to get version info
997 * cJSON *json = cJSON_Parse(response->response_data);
998 * // ... extract version fields ...
999 * cJSON_Delete(json);
1000 * obsws_response_free(response);
1001 * }
1002 *
1003 * @example Setting a source name (with parameters):
1004 * obsws_response_t *response = NULL;
1005 * const char *params = "{\"sourceName\": \"OldName\", \"newName\": \"NewName\"}";
1006 * obsws_send_request(conn, "SetSourceName", params, &response, 5000);
1007 * if (response) {
1008 * printf("Success: %d\n", response->success);
1009 * obsws_response_free(response);
1010 * }
1011 */
1012obsws_error_t obsws_send_request(obsws_connection_t *conn, const char *request_type,
1013 const char *request_data, obsws_response_t **response, uint32_t timeout_ms);
